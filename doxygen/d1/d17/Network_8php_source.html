216  public function processBatch(BatchPacket $packet, Player $p){
217  $str = zlib_decode($packet->payload, 1024 * 1024 * 64); //Max 64MB
218  $len = strlen($str);
219  $offset = 0;
220  try{
221  while($offset < $len){
222  if(($pk = $this->getPacket(ord($str{$offset++}))) !== null){
223  if($pk::NETWORK_ID === Info::BATCH_PACKET){
224  throw new \InvalidStateException("Invalid BatchPacket inside BatchPacket");
225  }
226  $pk->setBuffer($str, $offset);
227  $pk->decode();
228  $p->handleDataPacket($pk);
229  $offset += $pk->getOffset();
230  if($pk->getOffset() <= 0){
231  return;
232  }
233  }
234  }
235  }catch(\Exception $e){
236  if(\pocketmine\DEBUG > 1){
237  $logger = $this->server->getLogger();
238  if($logger instanceof MainLogger){
239  $logger->debug("BatchPacket " . " 0x" . bin2hex($packet->payload));
240  $logger->logException($e);
241  }
242  }
243  }
