140  if($currentOpt === null){
141  $output->plain[] = $currentLongString;
142  }else{
143  $output->opts[$currentOpt] = $currentLongString;
144  }
145  $currentLongString = "";
146  }
147  }elseif($arg{0} === '"'){ // start quote on
148  $quotesOn = true;
149  $currentLongString = substr($arg, 1);
150  }elseif($currentOpt !== null){
151  $output->opts[$currentOpt] = $arg;
152  $currentOpt = null;
153  }elseif($arg{0} === "."){
154  $output->switches[$arg] = true;
155  }elseif($arg{0} === ","){
156  $currentOpt = $arg;
157  }else{
158  $output->plain[] = $arg;
159  }
160  }
161  if($currentOpt !== null or $currentLongString !== "" or $quotesOn){
162  $output->unterminated = true;
163  }
