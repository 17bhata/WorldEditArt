74  public final function onRun(){

79  $this->onPostQuery($db);
80  if($result === false){

83  $resultType = $this->getResultType();
84  if($resultType === self::QUERY_NEGLECT){
85  if($result instanceof mysqli_result){
86  $result->close();
87  }
88  return;
89  }
90  if($resultType === self::QUERY_INSERT){
91  if($result instanceof mysqli_result){
92  $result->close();
93  throw new Exception("Internal code error: QUERY_INSERT must not be used with SELECT queries");
94  }
95  $this->insertId = $db->insert_id;
96  return;
97  }
98  if($resultType === self::QUERY_FETCH_ASSOC){
99  if(!($result instanceof mysqli_result)){
100  throw new Exception("Internal code error: QUERY_FETCH_ASSOC must be used with SELECT queries");
101  }
102  $this->assocRow = $result->fetch_assoc();
103  if(!is_array($this->assocRow)){
104  $this->assocNoResult = true;
105  throw new Exception("Unexpected value error: Empty result set");
106  }
107  $this->processRow($this->assocRow);
108  }elseif($resultType === self::QUERY_FETCH_ALL){
109  if(!($result instanceof mysqli_result)){
110  throw new Exception("Internal code error: QUERY_FETCH_ALL must be used with SELECT queries");
111  }
112  while(is_array($row = $result->fetch_assoc())){
113  $this->processRow($row);
114  $this->allRows[] = $row;
115  }
116  }
117  }catch(Exception $e){
