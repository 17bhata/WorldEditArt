161  public static function processArgs(array $args){

163  $output->plain = [];
164  $output->switches = [];
165  $output->opts = [];
166  $currentOpt = null;

172  if(substr($arg, -1) === '"'){
173  $currentLongString = substr($currentLongString, -1);
174  $quotesOn = false;
175  if($currentOpt === null){
176  $output->plain[] = $currentLongString;
177  }else{
178  $output->opts[$currentOpt] = $currentLongString;
179  }
180  $currentLongString = "";
181  }
182  }elseif($arg{0} === '"'){ // start quote on
183  $quotesOn = true;
184  $currentLongString = substr($arg, 1);
185  }elseif($currentOpt !== null){
186  $output->opts[$currentOpt] = $arg;
187  $currentOpt = null;
188  }elseif($arg{0} === "."){
189  $output->switches[$arg] = true;
190  }elseif($arg{0} === ","){
191  $currentOpt = $arg;
192  }else{
193  $output->plain[] = $arg;
194  }
195  }
196  if($currentOpt !== null or $currentLongString !== "" or $quotesOn){
197  $output->unterminated = true;
198  }
199  return $output;
200  }
