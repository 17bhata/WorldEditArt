175  function detect_system_timezone(){
176  switch(Utils::getOS()){
177  case 'win':
178  $regex = '/(UTC)(\+*\-*\d*\d*\:*\d*\d*)/';
179 
180  /*
181  * wmic timezone get Caption
182  * Get the timezone offset
183  *
184  * Sample Output var_dump
185  * array(3) {
186  * [0] =>
187  * string(7) "Caption"
188  * [1] =>
189  * string(20) "(UTC+09:30) Adelaide"
190  * [2] =>
191  * string(0) ""
192  * }
193  */
194  exec("wmic timezone get Caption", $output);
195 
196  $string = trim(implode("\n", $output));
197 
198  //Detect the Time Zone string
199  preg_match($regex, $string, $matches);
200 
201  if(!isset($matches[2])){
202  return false;
203  }
204 
205  $offset = $matches[2];
206 
207  if($offset == ""){
208  return "UTC";
209  }
210 
211  return parse_offset($offset);
212  break;
213  case 'linux':
214  // Ubuntu / Debian.
215  if(file_exists('/etc/timezone')){
216  $data = file_get_contents('/etc/timezone');
217  if($data){
218  return trim($data);
219  }
220  }
221

223  if(file_exists('/etc/sysconfig/clock')){
224  $data = parse_ini_file('/etc/sysconfig/clock');
225  if(!empty($data['ZONE'])){
226  return trim($data['ZONE']);
227  }
228  }
229 
230  //Portable method for incompatible linux distributions.
231 
232  $offset = trim(exec('date +%:z'));

235  return "UTC";

237 
238  return parse_offset($offset);

264  if(strpos($offset, '-') !== false){
