180  $z = (1 - $rainfall * $temperature) * 255;
181  $c = self::interpolateColor(256, $x, $z, [0x47, 0xd0, 0x33], [0x6c, 0xb4, 0x93], [0xbf, 0xb6, 0x55], [0x80, 0xb4, 0x97]);
182  return ((int) ($c[0] << 16)) | (int) (($c[1] << 8)) | (int) ($c[2]);
