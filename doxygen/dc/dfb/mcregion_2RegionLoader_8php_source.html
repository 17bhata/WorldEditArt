226  private function cleanGarbage(){
227  $sectors = [];
228  foreach($this->locationTable as $index => $data){ //Calculate file usage
229  if($data[0] === 0 or $data[1] === 0){
230  $this->locationTable[$index] = [0, 0, 0];
231  continue;
232  }
233  for($i = 0; $i < $data[1]; ++$i){
234  $sectors[$data[0]] = $index;
235  }
236  }
237 
238  if(count($sectors) === ($this->lastSector - 2)){ //No collection needed
239  return 0;
240  }
241 
242  ksort($sectors);
243  $shift = 0;
244  $lastSector = 1; //First chunk - 1
245 
246  fseek($this->filePointer, 8192);
247  $sector = 2;
248  foreach($sectors as $sector => $index){
249  if(($sector - $lastSector) > 1){
250  $shift += $sector - $lastSector - 1;
251  }
252  if($shift > 0){
253  fseek($this->filePointer, $sector << 12);
254  $old = fread($this->filePointer, 4096);
255  fseek($this->filePointer, ($sector - $shift) << 12);
256  fwrite($this->filePointer, $old, 4096);
257  }
258  $this->locationTable[$index][0] -= $shift;
259  $lastSector = $sector;
260  }
261  ftruncate($this->filePointer, ($sector + 1) << 12); //Truncate to the end of file written
262  return $shift;
263  }
